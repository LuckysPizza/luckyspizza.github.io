# (on gnu/linux) this example downloads zandronum, unpacks it, and sets up doom for you
$ wget https://zandronum.com/downloads/zandronum3.0-linux-x86_64.tar.bz2
$ tar -xvf zandronum3.0-linux-x86_64.tar.bz2
$ rm zandronum3.0-linux-x86_64.tar.bz2
$ mkdir -p ~/.config/zandronum && cd ~/.config/zandronum
$ wget https://luckyspizza.github.io/doom/f/doom.wad
$ exec ~/zandronum/zandronum -iwad doom
